```python
import torch
import torch.nn as nn
import triton
import triton.language as tl


@triton.jit
def matmul_kernel(
    A_ptr,  # Pointer to input tensor A
    B_ptr,  # Pointer to input tensor B
    C_ptr,  # Pointer to output tensor C
    b: tl.constexpr,  # Batch size
    i: tl.constexpr,  # Dimension i
    j: tl.constexpr,  # Dimension j
    l: tl.constexpr,  # Dimension l
    k: tl.constexpr,  # Dimension k
    BLOCK_SIZE: tl.constexpr,
):
    # Each program handles a contiguous block of data of size BLOCK_SIZE
    block_idx = tl.program_id(0)
    block_i = block_idx // (j * k)
    block_jk = block_idx % (j * k)
    block_j = block_jk // k
    block_k = block_jk % k

    # Create a range of offsets [0..BLOCK_SIZE-1]
    offsets_i = block_i * i + tl.arange(0, i)
    offsets_j = block_j * BLOCK_SIZE + tl.arange(0, BLOCK_SIZE)
    offsets_k = block_k * BLOCK_SIZE + tl.arange(0, BLOCK_SIZE)

    # Mask to ensure we don't go out of bounds
    mask_i = offsets_i < b * i
    mask_j = offsets_j < j
    mask_k = offsets_k < k

    # Load input values
    A = tl.load(A_ptr + offsets_i[:, None, None] * l + offsets_j[None, :, None] * 1 + offsets_k[None, None, :], mask=mask_i[:, None, None] & mask_j[None, :, None] & mask_k[None, None, :], other=0.0)
    B = tl.load(B_ptr + offsets_k[:, None] * k + offsets_j[None, :], mask=mask_k[:, None] & mask_j[None, :], other=0.0)

    # Initialize C to 0
    C = tl.zeros((i, j, k), dtype=tl.float32)

    # Perform the matmul
    for _ in range(BLOCK_SIZE):
        C += A * B

    # Store the result
    tl.store(C_ptr + offsets_i[:, None, None] * j * k + offsets_j[None, :, None] * k + offsets_k[None, None, :], C, mask=mask_i[:, None, None] & mask_j[None, :, None] & mask_k[None, None, :])


def triton_matmul(A: torch.Tensor, B: torch.Tensor):
    """
    This function wraps the Triton kernel call. It:
      1. Ensures the inputs are contiguous on GPU.
      2. Calculates the grid (blocks) needed.
      3. Launches the Triton kernel.
    """
    assert A.is_cuda and B.is_cuda, "Tensors must be on CUDA."
    A = A.contiguous()
    B = B.contiguous()

    # Prepare output tensor
    C = torch.empty((A.shape[0], A.shape[1], B.shape[1]), dtype=torch.float32, device=A.device)

    # Number of elements in the tensor
    b = A.shape[0]
    i = A.shape[1]
    j = B.shape[1]
    l = A.shape[3]
    k = B.shape[1]
    BLOCK_SIZE = 32  # Tunable parameter for block size

    # Determine the number of blocks needed
    grid = lambda meta: ((b * i + meta["BLOCK_SIZE"] - 1) // meta["BLOCK_SIZE"],)

    # Launch the Triton kernel
    matmul_kernel[grid](A, B, C, b, i, j, l, k, BLOCK_SIZE=BLOCK_SIZE)
    return C


class ModelNew(nn.Module):
    def __init__(self):
        super(ModelNew, self).__init__()

    def forward(self, A, B):
        return triton_matmul(A, B)
```
