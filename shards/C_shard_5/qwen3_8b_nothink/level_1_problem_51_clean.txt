import torch
import torch.nn as nn
import triton
import triton.language as tl


@triton.jit
def argmax_kernel(
    x_ptr,  # Pointer to input tensor
    out_ptr,  # Pointer to output tensor
    n_elements,  # Total number of elements in the input tensor
    dim: tl.constexpr,  # Dimension to compute argmax over
    BLOCK_SIZE: tl.constexpr,
):
    # Compute the index of the current program in the grid
    pid = tl.program_id(0)
    # Compute the offset for the current program
    offset = pid * BLOCK_SIZE
    # Compute the start and end indices for the current block
    start = offset
    end = start + BLOCK_SIZE
    # Compute the size of the tensor along the given dimension
    size = tl.load(x_ptr + (dim * n_elements) + (start // n_elements) * n_elements, other=0)
    # Compute the number of elements per block
    elements_per_block = tl.min(BLOCK_SIZE, size - start % size)
    # Compute the block's start and end indices along the given dimension
    block_start = start % size
    block_end = block_start + elements_per_block
    # Compute the indices for the current block
    idx = tl.arange(0, elements_per_block)
    # Compute the linear index for the current element
    linear_idx = (block_start + idx) * n_elements + tl.arange(0, n_elements)
    # Load the values
    x = tl.load(x_ptr + linear_idx, mask=linear_idx < n_elements, other=-float('inf'))
    # Compute the argmax
    max_val = tl.max(x)
    max_idx = tl.argmax(x)
    # Compute the index in the output tensor
    output_idx = (block_start + max_idx) // n_elements
    # Store the result
    tl.store(out_ptr + output_idx, output_idx)


def triton_argmax(x: torch.Tensor, dim: int):
    """
    This function wraps the Triton kernel call. It:
      1. Ensures the input is contiguous on GPU.
      2. Calculates the grid (blocks) needed.
      3. Launches the Triton kernel.
    """
    assert x.is_cuda, "Tensor must be on CUDA."
    x = x.contiguous()

    # Compute the number of elements in the input tensor
    n_elements = x.numel()
    # Compute the size of the tensor along the given dimension
    size = x.size(dim)
    # Compute the number of blocks needed
    BLOCK_SIZE = 128  # Tunable parameter for block size
    num_blocks = (size + BLOCK_SIZE - 1) // BLOCK_SIZE

    # Prepare output tensor
    out = torch.empty(size, dtype=torch.int64, device=x.device)

    # Launch the Triton kernel
    argmax_kernel[ num_blocks ](x, out, n_elements, dim, BLOCK_SIZE=BLOCK_SIZE)
    return out


class ModelNew(nn.Module):
    def __init__(self, dim: int):
        super(ModelNew, self).__init__()
        self.dim = dim

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return triton_argmax(x, self.dim)
