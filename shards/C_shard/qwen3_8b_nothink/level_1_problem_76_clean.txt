import torch
import torch.nn as nn
import triton
import triton.language as tl


@triton.jit
def conv1d_kernel(
    input_ptr,  # Pointer to input tensor (batch, in_channels, length)
    weight_ptr,  # Pointer to weight tensor (out_channels, in_channels, kernel_size)
    output_ptr,  # Pointer to output tensor (batch, out_channels, length_out)
    batch_size: tl.constexpr,
    in_channels: tl.constexpr,
    out_channels: tl.constexpr,
    kernel_size: tl.constexpr,
    stride: tl.constexpr,
    dilation: tl.constexpr,
    BLOCK_BATCH: tl.constexpr,
    BLOCK_IN: tl.constexpr,
    BLOCK_OUT: tl.constexpr,
    BLOCK_KERNEL: tl.constexpr,
):
    # Compute the batch, output channel, and input channel indices
    pid_batch = tl.program_id(0)
    pid_out = tl.program_id(1)
    pid_in = tl.program_id(2)

    # Compute the output length
    input_length = tl.load(input_ptr + tl.make_index(0, 0, 0), other=0)
    output_length = (input_length + dilation * (kernel_size - 1) - stride) // stride + 1

    # Compute the offset for the current batch, output channel, and input channel
    batch_offset = pid_batch * BLOCK_BATCH * in_channels * output_length
    out_offset = pid_out * BLOCK_OUT
    in_offset = pid_in * BLOCK_IN

    # Iterate over the output positions
    for out_pos in range(output_length):
        # Compute the input positions that contribute to this output position
        start_input_pos = out_pos * stride - (dilation * (kernel_size - 1))
        for kernel_pos in range(kernel_size):
            input_pos = start_input_pos + kernel_pos * dilation
            if input_pos < 0 or input_pos >= input_length:
                continue

            # Compute the indices for input, weight, and output
            input_idx = (batch_offset + out_pos * in_channels + pid_in) * input_length + input_pos
            weight_idx = (pid_out * in_channels + pid_in) * kernel_size + kernel_pos
            output_idx = (pid_batch * out_channels + pid_out) * output_length + out_pos

            # Load input and weight values
            input_val = tl.load(input_ptr + input_idx, other=0.0)
            weight_val = tl.load(weight_ptr + weight_idx, other=0.0)

            # Compute the output value
            output_val = input_val * weight_val

            # Accumulate the output value
            tl.atomic_add(output_ptr + output_idx, output_val)

    return


def triton_conv1d(input: torch.Tensor, weight: torch.Tensor, bias: torch.Tensor, stride: int, dilation: int):
    """
    This function wraps the Triton kernel call. It:
      1. Ensures the inputs are contiguous on GPU.
      2. Calculates the grid (blocks) needed.
      3. Launches the Triton kernel.
    """
    assert input.is_cuda and weight.is_cuda and bias.is_cuda, "Tensors must be on CUDA."
    input = input.contiguous()
    weight = weight.contiguous()
    bias = bias.contiguous()

    # Compute output dimensions
    batch_size = input.size(0)
    in_channels = input.size(1)
    input_length = input.size(2)
    out_channels = weight.size(0)
    kernel_size = weight.size(2)
    output_length = (input_length + dilation * (kernel_size - 1) - stride) // stride + 1

    # Prepare output tensor
    output = torch.empty(batch_size, out_channels, output_length, dtype=input.dtype, device=input.device)

    # Define block sizes
    BLOCK_BATCH = 16
    BLOCK_IN = 16
    BLOCK_OUT = 16
    BLOCK_KERNEL = 16

    # Determine the number of blocks needed
    grid = (  # (batch, out, in)
        (batch_size + BLOCK_BATCH - 1) // BLOCK_BATCH,
        (out_channels + BLOCK_OUT - 1) // BLOCK_OUT,
        (in_channels + BLOCK_IN - 1) // BLOCK_IN
    )

    # Launch the Triton kernel
    conv1d_kernel[grid](
        input, weight, output,
        batch_size, in_channels, out_channels, kernel_size, stride, dilation,
        BLOCK_BATCH, BLOCK_IN, BLOCK_OUT, BLOCK_KERNEL
    )

    # Add bias if present
    if bias is not None:
        output += bias.view(1, out_channels, 1)

    return output


class ModelNew(nn.Module):
    """
    Performs a standard 1D convolution operation with asymmetric input and a square kernel, potentially dilated and strided.
    This version uses a custom Triton kernel for optimized performance.
    """
    def __init__(self, in_channels: int, out_channels: int, kernel_size: int, stride: int = 1, dilation: int = 1, bias: bool = False):
        super(ModelNew, self).__init__()
        self.in_channels = in_channels
        self.out_channels = out_channels
        self.kernel_size = kernel_size
        self.stride = stride
        self.dilation = dilation
        self.bias = bias

        # Initialize weight and bias
        self.weight = nn.Parameter(torch.randn(out_channels, in_channels, kernel_size))
        if bias:
            self.bias = nn.Parameter(torch.randn(out_channels))
        else:
            self.register_parameter('bias', None)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        """
        Performs the 1D convolution using a custom Triton kernel.

        Args:
            x (torch.Tensor): Input tensor of shape (batch_size, in_channels, length).

        Returns:
            torch.Tensor: Output tensor of shape (batch_size, out_channels, length_out).
        """
        return triton_conv1d(x, self.weight, self.bias, self.stride, self.dilation)
