import torch
import torch.nn as nn
import triton
import triton.language as tl


@triton.jit
def min_kernel(
    x_ptr,  # Pointer to input tensor
    out_ptr,  # Pointer to output tensor
    n_elements,  # Total number of elements in the tensor
    dim,  # Dimension to reduce over
    BLOCK_SIZE: tl.constexpr,
    STRIDE: tl.constexpr,
):
    # Each program handles a contiguous block of data of size BLOCK_SIZE
    pid = tl.program_id(0)
    block_start = pid * BLOCK_SIZE
    offsets = block_start + tl.arange(0, BLOCK_SIZE)
    mask = offsets < n_elements

    # Load data
    x = tl.load(x_ptr + offsets, mask=mask, other=tl.max_float)

    # Compute the min along the specified dimension
    if dim == 0:
        # Reduce over the first dimension (batch)
        # Each element is a row, we need to find the min in each row
        # We'll compute the min across the row
        # For each row, we'll have to iterate through all elements
        # This is a naive approach and can be optimized further
        # But for simplicity, we'll compute the min for each row
        # Note: This is a simplified version and may not be optimal for all cases
        # We'll assume that the row is contiguous
        row_idx = tl.arange(0, BLOCK_SIZE)
        min_val = tl.min(x, axis=0)
        tl.store(out_ptr + row_idx, min_val, mask=row_idx < BLOCK_SIZE)
    elif dim == 1:
        # Reduce over the second dimension (features)
        # Each element is a column, we need to find the min in each column
        # We'll compute the min across the column
        # For each column, we'll have to iterate through all elements
        # This is a naive approach and can be optimized further
        # But for simplicity, we'll compute the min for each column
        # Note: This is a simplified version and may not be optimal for all cases
        # We'll assume that the column is contiguous
        col_idx = tl.arange(0, BLOCK_SIZE)
        min_val = tl.min(x, axis=1)
        tl.store(out_ptr + col_idx, min_val, mask=col_idx < BLOCK_SIZE)
    else:
        # For other dimensions, we need to handle it differently
        # This is a simplified version and may not be optimal for all cases
        # For the purpose of this example, we'll assume dim is 0 or 1
        pass


def triton_min(x: torch.Tensor, dim: int):
    """
    This function wraps the Triton kernel call. It:
      1. Ensures the inputs are contiguous on GPU.
      2. Calculates the grid (blocks) needed.
      3. Launches the Triton kernel.
    """
    assert x.is_cuda, "Tensors must be on CUDA."
    x = x.contiguous()

    # Prepare output tensor
    out_shape = list(x.shape)
    out_shape[dim] = 1
    out = torch.empty(out_shape, dtype=x.dtype, device=x.device)

    # Number of elements in the tensor
    n_elements = x.numel()
    BLOCK_SIZE = 128  # Tunable parameter for block size

    # Determine the number of blocks needed
    grid = lambda meta: ((n_elements + meta["BLOCK_SIZE"] - 1) // meta["BLOCK_SIZE"],)

    # Launch the Triton kernel
    min_kernel[grid](x, out, n_elements, dim, BLOCK_SIZE=BLOCK_SIZE, STRIDE=x.stride(0))
    return out


class ModelNew(nn.Module):
    """
    Optimized model that performs min reduction over a specific dimension using a custom Triton kernel.
    """
    def __init__(self, dim: int):
        """
        Initializes the model with the dimension to reduce over.

        Args:
            dim (int): The dimension to reduce over.
        """
        super(ModelNew, self).__init__()
        self.dim = dim

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        """
        Applies min reduction over the specified dimension to the input tensor using a custom Triton kernel.

        Args:
            x (torch.Tensor): Input tensor.

        Returns:
            torch.Tensor: Output tensor after min reduction over the specified dimension.
        """
        return triton_min(x, self.dim)
