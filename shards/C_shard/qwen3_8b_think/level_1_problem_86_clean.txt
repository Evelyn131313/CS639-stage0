import torch
import torch.nn as nn
import triton
import triton.language as tl


@triton.jit
def depthwise_conv_kernel(
    input_ptr, 
    weight_ptr, 
    output_ptr, 
    batch_size, 
    in_channels, 
    out_channels, 
    kernel_size, 
    stride, 
    padding, 
    dilation, 
    BLOCK_SIZE: tl.constexpr
):
    pid = tl.program_id(0)
    block_start = pid * BLOCK_SIZE
    offsets = block_start + tl.arange(0, BLOCK_SIZE)
    mask = offsets < (batch_size * in_channels * (height * width))
    input = tl.load(input_ptr + offsets, mask=mask, other=0.0)
    weight = tl.load(weight_ptr + offsets, mask=mask, other=0.0)
    output = tl.dot(input, weight)
    tl.store(output_ptr + offsets, output, mask=mask)


@triton.jit
def pointwise_conv_kernel(
    input_ptr, 
    weight_ptr, 
    output_ptr, 
    batch_size, 
    in_channels, 
    out_channels, 
    height, 
    width, 
    BLOCK_SIZE: tl.constexpr
):
    pid = tl.program_id(0)
    block_start = pid * BLOCK_SIZE
    offsets = block_start + tl.arange(0, BLOCK_SIZE)
    mask = offsets < (batch_size * height * width * out_channels)
    input = tl.load(input_ptr + offsets, mask=mask, other=0.0)
    weight = tl.load(weight_ptr + offsets, mask=mask, other=0.0)
    output = tl.dot(input, weight)
    tl.store(output_ptr + offsets, output, mask=mask)


def triton_depthwise_conv(x: torch.Tensor, weight: torch.Tensor, bias: torch.Tensor, kernel_size, stride, padding, dilation):
    assert x.is_cuda and weight.is_cuda, "Tensors must be on CUDA."
    x = x.contiguous()
    weight = weight.contiguous()
    output = torch.empty_like(x)
    n_elements = x.numel()
    BLOCK_SIZE = 128
    grid = lambda meta: ((n_elements + meta["BLOCK_SIZE"] - 1) // meta["BLOCK_SIZE"],)
    depthwise_conv_kernel[grid](x, weight, output, x.size(0), x.size(1), x.size(2), kernel_size, stride, padding, dilation, BLOCK_SIZE=BLOCK_SIZE)
    return output


def triton_pointwise_conv(x: torch.Tensor, weight: torch.Tensor, bias: torch.Tensor):
    assert x.is_cuda and weight.is_cuda, "Tensors must be on CUDA."
    x = x.contiguous()
    weight = weight.contiguous()
    output = torch.empty_like(x)
    n_elements = x.numel()
    BLOCK_SIZE = 128
    grid = lambda meta: ((n_elements + meta["BLOCK_SIZE"] - 1) // meta["BLOCK_SIZE"],)
    pointwise_conv_kernel[grid](x, weight, output, x.size(0), x.size(1), x.size(2), x.size(3), x.size(3), BLOCK_SIZE=BLOCK_SIZE)
    return output


class ModelNew(nn.Module):
    def __init__(self, in_channels: int, out_channels: int, kernel_size: int, stride: int = 1, padding: int = 0, dilation: int = 1, bias: bool = False):
        super(ModelNew, self).__init__()
        self.depthwise_weight = nn.Parameter(torch.randn(in_channels, in_channels, kernel_size, kernel_size))
        self.pointwise_weight = nn.Parameter(torch.randn(out_channels, in_channels))
        self.bias = nn.Parameter(torch.randn(out_channels)) if bias else None

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        x = triton_depthwise_conv(x, self.depthwise_weight, self.bias, self.depthwise_weight.size(2), self.depthwise_weight.size(2), self.depthwise_weight.size(2), self.depthwise_weight.size(2))
        x = triton_pointwise_conv(x, self.pointwise_weight, self.bias)
        return x
