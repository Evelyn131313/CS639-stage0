import torch
import torch.nn as nn
import triton
import triton.language as tl


@triton.jit
def conv1d_kernel(
    x_ptr,  # Pointer to input tensor
    w_ptr,  # Pointer to weight tensor
    y_ptr,  # Pointer to output tensor
    x_shape,  # Shape of input tensor (batch_size, in_channels, height, width)
    w_shape,  # Shape of weight tensor (out_channels, in_channels, kernel_size)
    y_shape,  # Shape of output tensor (batch_size, out_channels, height, width)
    BLOCK_SIZE: tl.constexpr,
    GROUP_SIZE: tl.constexpr,
):
    # Each program handles a contiguous block of data of size BLOCK_SIZE
    batch_id = tl.program_id(0)
    out_channel_id = tl.program_id(1)

    # Calculate the height and width indices
    height_id = tl.program_id(2)
    width_id = tl.program_id(3)

    # Calculate the offsets
    x_offset = batch_id * x_shape[1] * x_shape[2] * x_shape[3] + out_channel_id * x_shape[2] * x_shape[3] + height_id * x_shape[3] + width_id
    w_offset = out_channel_id * w_shape[1] * w_shape[2] + (width_id % GROUP_SIZE) * w_shape[2] + (width_id // GROUP_SIZE)
    y_offset = batch_id * y_shape[1] * y_shape[2] * y_shape[3] + out_channel_id * y_shape[2] * y_shape[3] + height_id * y_shape[3] + width_id

    # Initialize output to zero
    y = tl.zeros((), dtype=tl.float32)

    # Load input and weight values
    x = tl.load(x_ptr + x_offset, mask=width_id < x_shape[3], other=0.0)
    w = tl.load(w_ptr + w_offset, mask=width_id < w_shape[2], other=0.0)

    # Perform the convolution
    for k in range(w_shape[2]):
        y += x * w[k]

    # Store the result
    tl.store(y_ptr + y_offset, y)


class ModelNew(nn.Module):
    def __init__(self, in_channels: int, out_channels: int, bias: bool = False):
        super(ModelNew, self).__init__()
        self.conv1d = nn.Conv2d(in_channels, out_channels, kernel_size=1, stride=1, padding=0, bias=bias)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        # Convert the PyTorch convolution to Triton
        batch_size = x.shape[0]
        in_channels = x.shape[1]
        height = x.shape[2]
        width = x.shape[3]
        out_channels = self.conv1d.out_channels
        kernel_size = self.conv1d.kernel_size[0]

        # Prepare output tensor
        y = torch.empty((batch_size, out_channels, height, width), device=x.device, dtype=x.dtype)

        # Launch the Triton kernel
        BLOCK_SIZE = 128  # Tunable parameter for block size
        GROUP_SIZE = 16   # Tunable parameter for group size

        grid = lambda meta: (
            batch_size,
            out_channels,
            (height + meta["BLOCK_SIZE"] - 1) // meta["BLOCK_SIZE"],
            (width + meta["BLOCK_SIZE"] - 1) // meta["BLOCK_SIZE"],
        )

        conv1d_kernel[grid](x, self.conv1d.weight, y, x.shape, self.conv1d.weight.shape, y.shape, BLOCK_SIZE=BLOCK_SIZE, GROUP_SIZE=GROUP_SIZE)

        return y
