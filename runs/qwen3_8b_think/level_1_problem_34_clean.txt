def triton_instance_norm(x: torch.Tensor, gamma: torch.Tensor, beta: torch.Tensor, eps: float = 1e-5):
    """
    This function wraps the Triton kernel call. It:
      1. Ensures the inputs are contiguous on GPU.
      2. Calculates the grid (blocks) needed.
      3. Launches the Triton kernel.
    """
    assert x.is_cuda and gamma.is_cuda and beta.is_cuda, "Tensors must be on CUDA."
    x = x.contiguous()
    gamma = gamma.contiguous()
    beta = beta.contiguous()

    # Prepare output tensor
    out = torch.empty_like(x)

    # Number of elements in the tensor
    n_elements = x.numel()
    BLOCK_SIZE = 128  # Tunable parameter for block size

    # Determine the number of blocks needed
    grid = lambda meta: ((n_elements + meta["BLOCK_SIZE"] - 1) // meta["BLOCK_SIZE"],)

    # Launch the Triton kernel
    instance_norm_kernel[grid](x, gamma, beta, out, x.size(0), x.size(1), x.size(2), x.size(3), eps, BLOCK_SIZE=BLOCK_SIZE)
    return out


class ModelNew(nn.Module):
    def __init__(self, num_features: int):

        self.num_features = num_features

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        # Create learnable parameters for gamma and beta
        gamma = nn.Parameter(torch.randn(self.num_features, device=x.device))
        beta = nn.Parameter(torch.randn(self.num_features, device=x.device))
        return triton_instance_norm(x, gamma, beta)
