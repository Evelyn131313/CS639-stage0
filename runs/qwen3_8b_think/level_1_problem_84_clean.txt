import torch
import torch.nn as nn
import triton
import triton.language as tl


@triton.jit
def depthwise_conv2d_kernel(
    input_ptr,  # Pointer to input tensor (batch, in_channels, height, width)
    weight_ptr,  # Pointer to weight tensor (out_channels, in_channels, kernel_size, kernel_size)
    output_ptr,  # Pointer to output tensor (batch, out_channels, height_out, width_out)
    batch, in_channels, out_channels, height, width, kernel_size, stride, padding,
    BLOCK_SIZE: tl.constexpr,
):
    # Each thread handles a single output element
    # We use 2D thread indexing for spatial dimensions
    pid_x = tl.program_id(0)
    pid_y = tl.program_id(1)
    # Compute the output position
    out_h = pid_x
    out_w = pid_y
    # Compute the input position range
    input_h_start = out_h * stride - padding
    input_w_start = out_w * stride - padding
    # Iterate over the kernel
    acc = tl.zeros((out_channels,), dtype=tl.float32)
    for k in range(kernel_size):
        for l in range(kernel_size):
            input_h = input_h_start + k
            input_w = input_w_start + l
            # Load input data for each channel
            for c in range(in_channels):
                input_val = tl.load(input_ptr + (out_h * width + out_w) * in_channels * height * width + (c * height * width + input_h * width + input_w))
                weight_val = tl.load(weight_ptr + (c * kernel_size * kernel_size + k * kernel_size + l))
                acc[c] += input_val * weight_val
    # Store the result
    tl.store(output_ptr + (out_h * width_out + out_w) * out_channels + c, acc[c])


def triton_depthwise_conv2d(x: torch.Tensor, weight: torch.Tensor, bias: torch.Tensor, stride: int, padding: int):
    """
    Custom Triton kernel for depthwise convolution.
    """
    assert x.is_cuda and weight.is_cuda, "Tensors must be on CUDA."
    x = x.contiguous()
    weight = weight.contiguous()
    output = torch.empty((x.size(0), weight.size(0), x.size(2) - kernel_size + 1, x.size(3) - kernel_size + 1), dtype=x.dtype, device=x.device)

    # Calculate output dimensions
    height_out = (x.size(2) + 2 * padding - kernel_size) // stride + 1
    width_out = (x.size(3) + 2 * padding - kernel_size) // stride + 1

    # Determine the number of blocks needed
    grid = (height_out, width_out)

    # Launch the Triton kernel
    depthwise_conv2d_kernel[grid](x, weight, output, x.size(0), x.size(1), weight.size(0), x.size(2), x.size(3), kernel_size, stride, padding, BLOCK_SIZE=128)
    return output


class ModelNew(nn.Module):
    def __init__(self, in_channels: int, out_channels: int, kernel_size: int, stride: int = 1, padding: int = 0):
        super(ModelNew, self).__init__()
        self.in_channels = in_channels
        self.out_channels = out_channels
        self.kernel_size = kernel_size
        self.stride = stride
        self.padding = padding
        # Initialize weights and biases
        self.weight = nn.Parameter(torch.randn(out_channels, in_channels, kernel_size, kernel_size))
        self.bias = nn.Parameter(torch.randn(out_channels))

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        # Perform depthwise convolution with custom Triton kernel
        return triton_depthwise_conv2d(x, self.weight, self.bias, self.stride, self.padding)
