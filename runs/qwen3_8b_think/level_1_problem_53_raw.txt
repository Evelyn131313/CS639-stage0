def min_reduction_kernel(

    out_ptr,  # Pointer to output tensor

    n_dim2,  # Size of dim2
    dim: tl.constexpr,  # Dimension to reduce
    BLOCK_SIZE: tl.constexpr,
):

    b = pid // n_dim2
    i = pid % n_dim2

        # Reduce along dim1 (second dimension)
        # Each (b, i) in output corresponds to (b, :, i) in input
