import torch
import torch.nn as nn
import triton
import triton.language as tl

@triton.jit
def conv1d_kernel(
    input_ptr,
    weights_ptr,
    output_ptr,
    batch_size,
    in_channels,
    out_channels,
    kernel_size,
    stride,
    dilation,
    input_length,
    output_length,
    BLOCK_SIZE: tl.constexpr,
):
    pid = tl.program_id(0)
    batch = pid // (out_channels * output_length)
    remaining = pid % (out_channels * output_length)
    out_channel = remaining // output_length
    out_pos = remaining % output_length

    start_input_pos = out_pos * stride
    input_positions = start_input_pos + tl.arange(0, kernel_size) * dilation
    mask = input_positions < input_length

    input_vals = tl.load(input_ptr + (batch * in_channels * input_length) + tl.arange(0, in_channels) * input_length + input_positions, mask=mask, other=0.0)
    weights_vals = tl.load(weights_ptr + (out_channel * in_channels * kernel_size) + tl.arange(0, in_channels) * kernel_size + tl.arange(0, kernel_size), mask=mask, other=0.0)

    sum_val = tl.sum(input_vals * weights_vals)
    tl.store(output_ptr + (batch * out_channels * output_length) + (out_channel * output_length) + out_pos, sum_val)

def triton_conv1d(x: torch.Tensor, weights: torch.Tensor, stride: int, dilation: int):
    assert x.is_cuda and weights.is_cuda, "Tensors must be on CUDA."
    x = x.contiguous()
    weights = weights.contiguous()

    batch_size = x.size(0)
    in_channels = x.size(1)
    input_length = x.size(2)
    out_channels = weights.size(0)
    kernel_size = weights.size(2)

    output_length = (input_length - (kernel_size - 1) * dilation - 1) // stride + 1

    output = torch.empty((batch_size, out_channels, output_length), dtype=x.dtype, device=x.device)

    grid = (batch_size * out_channels * output_length, 1, 1)

    conv1d_kernel[grid](x, weights, output, batch_size, in_channels, out_channels, kernel_size, stride, dilation, input_length, output_length, BLOCK_SIZE=128)

    return output

class ModelNew(nn.Module):
    def __init__(self, in_channels: int, out_channels: int, kernel_size: int, stride: int = 1, dilation: int = 1, bias: bool = False):
        super(ModelNew, self).__init__()
        self.in_channels = in_channels
        self.out_channels = out_channels
        self.kernel_size = kernel_size
        self.stride = stride
        self.dilation = dilation
        self.bias = bias
        self.weight = nn.Parameter(torch.randn(out_channels, in_channels, kernel_size).cuda())
        if bias:
            self.bias = nn.Parameter(torch.randn(out_channels).cuda())
        else:
            self.bias = None

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        output = triton_conv1d(x, self.weight, self.stride, self.dilation)
        if self.bias is not None:
            output += self.bias.view(1, -1, 1)
        return output
