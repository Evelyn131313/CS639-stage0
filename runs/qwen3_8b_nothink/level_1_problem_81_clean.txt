def triton_conv_transpose2d(input: torch.Tensor, weight: torch.Tensor, bias: torch.Tensor, stride: int, padding: int, dilation: int):
    """
    This function wraps the Triton kernel call for transposed convolution.
    """
    assert input.is_cuda and weight.is_cuda and bias.is_cuda, "Tensors must be on CUDA."
    input = input.contiguous()
    weight = weight.contiguous()
    bias = bias.contiguous()

    # Compute output dimensions
    batch_size = input.size(0)
    in_channels = input.size(1)
    in_h = input.size(2)
    in_w = input.size(3)
    out_channels = weight.size(0)
    kernel_size = weight.size(1)
    out_h = (in_h - 1) * stride - 2 * padding + dilation * (kernel_size - 1) + 1
    out_w = (in_w - 1) * stride - 2 * padding + dilation * (kernel_size - 1) + 1
    output = torch.empty((batch_size, out_channels, out_h, out_w), dtype=input.dtype, device=input.device)

    # Prepare output tensor
    n_elements = batch_size * out_channels * out_channels
    BLOCK_SIZE = 128  # Tunable parameter for block size

    # Determine the number of blocks needed
    grid = lambda meta: ((n_elements + meta["BLOCK_SIZE"] - 1) // meta["BLOCK_SIZE"],)

    # Launch the Triton kernel
    conv_transpose2d_kernel[grid](input, weight, output, batch_size, in_channels, out_channels, kernel_size, stride, padding, dilation, BLOCK_SIZE=BLOCK_SIZE)
    return output
