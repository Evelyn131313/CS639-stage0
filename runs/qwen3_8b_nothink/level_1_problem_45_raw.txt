import torch
import torch.nn as nn
import triton
import triton.language as tl


@triton.jit
def avg_pool_kernel(
    input_ptr,  # Pointer to input tensor
    output_ptr,  # Pointer to output tensor
    batch_size: tl.constexpr,
    channels: tl.constexpr,
    height: tl.constexpr,
    width: tl.constexpr,
    kernel_size: tl.constexpr,
    stride: tl.constexpr,
    padding: tl.constexpr,
    BLOCK_SIZE: tl.constexpr,
):
    # Each program processes a single output element
    pid = tl.program_id(0)
    # Compute the output position
    out_h = pid // (width // stride)
    out_w = pid % (width // stride)
    # Compute the input range for this output element
    in_h_start = out_h * stride - padding
    in_h_end = in_h_start + kernel_size
    in_w_start = out_w * stride - padding
    in_w_end = in_w_start + kernel_size

    # Compute the total number of elements in the kernel
    num_elements = kernel_size * kernel_size
    # Initialize sum and count
    sum_val = tl.zeros((num_elements,), dtype=tl.float32)
    count = tl.zeros((num_elements,), dtype=tl.int32)

    # Iterate over the kernel region
    for i in range(kernel_size):
        for j in range(kernel_size):
            in_h = in_h_start + i
            in_w = in_w_start + j
            if in_h < 0 or in_h >= height or in_w < 0 or in_w >= width:
                continue
            idx = (in_h * width + in_w)
            idx = (pid * width * height + idx)
            idx = (batch_size * channels * width * height + idx)
            val = tl.load(input_ptr + idx, mask=(in_h >= 0) & (in_h < height) & (in_w >= 0) & (in_w < width))
            sum_val[i * kernel_size + j] = val
            count[i * kernel_size + j] = 1

    # Compute average
    avg = sum_val / count
    # Store result
    idx = (out_h * width + out_w)
    idx = (batch_size * channels * width * height + idx)
    tl.store(output_ptr + idx, avg)


def triton_avg_pool(input: torch.Tensor, kernel_size: int, stride: int, padding: int):
    """
    Applies 2D Average Pooling using a custom Triton kernel.
    """
    assert input.is_cuda, "Input tensor must be on CUDA."
    batch_size = input.size(0)
    channels = input.size(1)
    height = input.size(2)
    width = input.size(3)
    output_height = (height + 2 * padding - kernel_size) // stride + 1
    output_width = (width + 2 * padding - kernel_size) // stride + 1

    output = torch.empty((batch_size, channels, output_height, output_width), dtype=input.dtype, device=input.device)

    # Determine block size
    BLOCK_SIZE = 128  # Tunable parameter for block size

    # Grid size: one block per output element
    grid = (output.numel(),)

    # Launch the Triton kernel
    avg_pool_kernel[grid](
        input, output, batch_size, channels, height, width, kernel_size, stride, padding, BLOCK_SIZE
    )
    return output


class ModelNew(nn.Module):
    """
    Optimized model using a custom Triton kernel for 2D Average Pooling.
    """
    def __init__(self, kernel_size: int, stride: int = None, padding: int = 0):
        super(ModelNew, self).__init__()
        self.kernel_size = kernel_size
        self.stride = stride if stride is not None else kernel_size
        self.padding = padding

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return triton_avg_pool(x, self.kernel_size, self.stride, self.padding)
