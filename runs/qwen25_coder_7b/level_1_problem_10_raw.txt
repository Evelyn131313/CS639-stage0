import torch
import torch.nn as nn
import triton
import triton.language as tl


@triton.jit
def matmul_kernel(
    A_ptr,  # Pointer to input tensor A
    B_ptr,  # Pointer to input tensor B
    C_ptr,  # Pointer to output tensor C
    N: tl.constexpr,  # Number of blocks in the first dimension of A
    M: tl.constexpr,  # Number of blocks in the second dimension of A
    K: tl.constexpr,  # Number of blocks in the third dimension of A and the first dimension of B
    L: tl.constexpr,  # Number of blocks in the second dimension of B
    BLOCK_SIZE: tl.constexpr,
):
    # Each program handles a contiguous block of data of size BLOCK_SIZE
    block_row = tl.program_id(0)
    block_col = tl.program_id(1)
    block_k = tl.program_id(2)

    # Calculate the row and column indices within the block
    row = block_row * BLOCK_SIZE + tl.arange(0, BLOCK_SIZE)
    col = block_col * BLOCK_SIZE + tl.arange(0, BLOCK_SIZE)
    k = block_k * BLOCK_SIZE + tl.arange(0, BLOCK_SIZE)

    # Initialize the result to zero
    C = tl.zeros((BLOCK_SIZE, BLOCK_SIZE), dtype=tl.float32)

    # Perform the matrix multiplication
    for i in range(BLOCK_SIZE):
        a = tl.load(A_ptr + (row[i] * K + k[i]) * BLOCK_SIZE, mask=tl.all(k < K), other=0.0)
        b = tl.load(B_ptr + (k[i] * L + col[i]) * BLOCK_SIZE, mask=tl.all(k < K), other=0.0)
        C += a * b

    # Store the result
    tl.store(C_ptr + (block_row * M + block_col) * BLOCK_SIZE * BLOCK_SIZE, C, mask=tl.all(row < N) & tl.all(col < L))


def triton_matmul(A: torch.Tensor, B: torch.Tensor):
    """
    This function wraps the Triton kernel call. It:
      1. Ensures the inputs are contiguous on GPU.
      2. Calculates the grid (blocks) needed.
      3. Launches the Triton kernel.
    """
    assert A.is_cuda and B.is_cuda, "Tensors must be on CUDA."
    A = A.contiguous()
    B = B.contiguous()

    # Prepare output tensor
    C = torch.empty((A.shape[0], A.shape[1], B.shape[1]), dtype=A.dtype, device=A.device)

    # Number of blocks in each dimension
    N = A.shape[0] // BLOCK_SIZE
    M = A.shape[1] // BLOCK_SIZE
    K = A.shape[2] // BLOCK_SIZE
    L = B.shape[1] // BLOCK_SIZE

    BLOCK_SIZE = 128  # Tunable parameter for block size

    # Determine the number of blocks needed
    grid = lambda meta: ((N + meta["BLOCK_SIZE"] - 1) // meta["BLOCK_SIZE"],
                          (M + meta["BLOCK_SIZE"] - 1) // meta["BLOCK_SIZE"],
                          (K + meta["BLOCK_SIZE"] - 1) // meta["BLOCK_SIZE"])

    # Launch the Triton kernel
    matmul_kernel[grid](A, B, C, N, M, K, L, BLOCK_SIZE=BLOCK_SIZE)
    return C


class ModelNew(nn.Module):
    def __init__(self):
        super(ModelNew, self).__init__()
    
    def forward(self, A, B):
        return triton_matmul(A, B)
