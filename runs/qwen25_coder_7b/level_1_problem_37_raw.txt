import torch
import torch.nn as nn
import triton
import triton.language as tl


@triton.jit
def frobenius_norm_kernel(
    x_ptr,  # Pointer to input tensor
    norm_ptr,  # Pointer to output norm
    n_elements,  # Total number of elements in input
    BLOCK_SIZE: tl.constexpr,
):
    # Each program handles a contiguous block of data of size BLOCK_SIZE
    block_start = tl.program_id(0) * BLOCK_SIZE
    # Create a range of offsets [0..BLOCK_SIZE-1]
    offsets = block_start + tl.arange(0, BLOCK_SIZE)
    # Mask to ensure we don't go out of bounds
    mask = offsets < n_elements
    # Load input values
    x = tl.load(x_ptr + offsets, mask=mask, other=0.0)
    # Compute the square of the elements
    x_sq = x * x
    # Reduce sum to get the norm squared
    norm_sq = tl.sum(x_sq, mask=mask)
    # Store the result
    tl.store(norm_ptr + block_start, norm_sq, mask=mask)


@triton.jit
def normalize_kernel(
    x_ptr,  # Pointer to input tensor
    norm_ptr,  # Pointer to input norm
    out_ptr,  # Pointer to output tensor
    n_elements,  # Total number of elements in input/output
    BLOCK_SIZE: tl.constexpr,
):
    # Each program handles a contiguous block of data of size BLOCK_SIZE
    block_start = tl.program_id(0) * BLOCK_SIZE
    # Create a range of offsets [0..BLOCK_SIZE-1]
    offsets = block_start + tl.arange(0, BLOCK_SIZE)
    # Mask to ensure we don't go out of bounds
    mask = offsets < n_elements
    # Load input values
    x = tl.load(x_ptr + offsets, mask=mask, other=0.0)
    norm = tl.load(norm_ptr + block_start, mask=mask, other=1.0)
    # Normalize the elements
    out = x / norm
    # Store the result
    tl.store(out_ptr + offsets, out, mask=mask)


def triton_frobenius_norm(x: torch.Tensor):
    """
    This function wraps the Triton kernel call for Frobenius norm computation. It:
      1. Ensures the input is contiguous on GPU.
      2. Calculates the grid (blocks) needed.
      3. Launches the Triton kernel.
    """
    assert x.is_cuda, "Tensor must be on CUDA."
    x = x.contiguous()

    # Prepare output tensor for norm
    norm = torch.empty(1, dtype=torch.float32, device=x.device)

    # Number of elements in the tensor
    n_elements = x.numel()
    BLOCK_SIZE = 128  # Tunable parameter for block size

    # Determine the number of blocks needed
    grid = lambda meta: ((n_elements + meta["BLOCK_SIZE"] - 1) // meta["BLOCK_SIZE"],)

    # Launch the Triton kernel for norm computation
    frobenius_norm_kernel[grid](x, norm, n_elements, BLOCK_SIZE=BLOCK_SIZE)
    return norm.sqrt()


def triton_normalize(x: torch.Tensor, norm: torch.Tensor):
    """
    This function wraps the Triton kernel call for normalization. It:
      1. Ensures the input and norm are contiguous on GPU.
      2. Calculates the grid (blocks) needed.
      3. Launches the Triton kernel.
    """
    assert x.is_cuda and norm.is_cuda, "Tensors must be on CUDA."
    x = x.contiguous()
    norm = norm.contiguous()

    # Prepare output tensor
    out = torch.empty_like(x)

    # Number of elements in the tensor
    n_elements = x.numel()
    BLOCK_SIZE = 128  # Tunable parameter for block size

    # Determine the number of blocks needed
    grid = lambda meta: ((n_elements + meta["BLOCK_SIZE"] - 1) // meta["BLOCK_SIZE"],)

    # Launch the Triton kernel for normalization
    normalize_kernel[grid](x, norm, out, n_elements, BLOCK_SIZE=BLOCK_SIZE)
    return out


class ModelNew(nn.Module):
    def __init__(self):
        super(ModelNew, self).__init__()

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        norm = triton_frobenius_norm(x)
        return triton_normalize(x, norm)
