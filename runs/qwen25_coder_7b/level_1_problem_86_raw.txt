import torch
import torch.nn as nn
import triton
import triton.language as tl


@triton.jit
def depthwise_conv2d_kernel(
    x_ptr,  # Pointer to input tensor
    weight_ptr,  # Pointer to depthwise convolution weight
    out_ptr,  # Pointer to output tensor
    n_batch, n_channels, height, width, kernel_size, stride, padding, dilation, 
    BLOCK_SIZE: tl.constexpr,
):
    # Each program handles a contiguous block of data of size BLOCK_SIZE
    block_start = tl.program_id(0) * BLOCK_SIZE
    # Create a range of offsets [0..BLOCK_SIZE-1]
    offsets = block_start + tl.arange(0, BLOCK_SIZE)
    # Mask to ensure we don't go out of bounds
    mask = offsets < n_batch * n_channels * height * width
    # Load input values
    x = tl.load(x_ptr + offsets, mask=mask, other=0.0)
    
    # Load weight values
    weight = tl.load(weight_ptr + offsets, mask=mask, other=0.0)
    
    # Calculate output indices
    b = offsets // (n_channels * height * width)
    c = (offsets // (height * width)) % n_channels
    y = (offsets // width) % height
    x_idx = offsets % width
    
    # Perform the depthwise convolution
    out = 0.0
    for ky in range(kernel_size):
        for kx in range(kernel_size):
            wy = y * stride + ky * dilation
            wx = x_idx * stride + kx * dilation
            if 0 <= wy < height and 0 <= wx < width:
                out += x[b, c, wy, wx] * weight[ky * kernel_size + kx]
    
    # Store the result
    tl.store(out_ptr + offsets, out, mask=mask)


@triton.jit
def pointwise_conv2d_kernel(
    x_ptr,  # Pointer to input tensor
    weight_ptr,  # Pointer to pointwise convolution weight
    out_ptr,  # Pointer to output tensor
    n_batch, n_channels, height, width, out_channels, 
    BLOCK_SIZE: tl.constexpr,
):
    # Each program handles a contiguous block of data of size BLOCK_SIZE
    block_start = tl.program_id(0) * BLOCK_SIZE
    # Create a range of offsets [0..BLOCK_SIZE-1]
    offsets = block_start + tl.arange(0, BLOCK_SIZE)
    # Mask to ensure we don't go out of bounds
    mask = offsets < n_batch * out_channels * height * width
    # Load input values
    x = tl.load(x_ptr + offsets, mask=mask, other=0.0)
    
    # Load weight values
    weight = tl.load(weight_ptr + offsets, mask=mask, other=0.0)
    
    # Calculate output indices
    b = offsets // (out_channels * height * width)
    c = (offsets // (height * width)) % out_channels
    y = (offsets // width) % height
    x_idx = offsets % width
    
    # Perform the pointwise convolution
    out = 0.0
    for ci in range(c):
        out += x[b, ci, y, x_idx] * weight[c * out_channels + ci]
    
    # Store the result
    tl.store(out_ptr + offsets, out, mask=mask)


def triton_depthwise_conv2d(x: torch.Tensor, weight: torch.Tensor):
    """
    This function wraps the Triton kernel call. It:
      1. Ensures the inputs are contiguous on GPU.
      2. Calculates the grid (blocks) needed.
      3. Launches the Triton kernel.
    """
    assert x.is_cuda and weight.is_cuda, "Tensors must be on CUDA."
    x = x.contiguous()
    weight = weight.contiguous()

    # Prepare output tensor
    out = torch.empty_like(x)

    # Number of elements in the tensor
    n_batch = x.shape[0]
    n_channels = x.shape[1]
    height = x.shape[2]
    width = x.shape[3]
    kernel_size = weight.shape[2]
    stride = 1
    padding = 0
    dilation = 1
    BLOCK_SIZE = 128  # Tunable parameter for block size

    # Determine the number of blocks needed
    grid = lambda meta: ((n_batch * n_channels * height * width + meta["BLOCK_SIZE"] - 1) // meta["BLOCK_SIZE"],)

    # Launch the Triton kernel
    depthwise_conv2d_kernel[grid](x, weight, out, n_batch, n_channels, height, width, kernel_size, stride, padding, dilation, BLOCK_SIZE=BLOCK_SIZE)
    return out


def triton_pointwise_conv2d(x: torch.Tensor, weight: torch.Tensor):
    """
    This function wraps the Triton kernel call. It:
      1. Ensures the inputs are contiguous on GPU.
      2. Calculates the grid (blocks) needed.
      3. Launches the Triton kernel.
    """
    assert x.is_cuda and weight.is_cuda, "Tensors must be on CUDA."
    x = x.contiguous()
    weight = weight.contiguous()

    # Prepare output tensor
    out = torch.empty_like(x)

    # Number of elements in the tensor
    n_batch = x.shape[0]
    n_channels = x.shape[1]
    height = x.shape[2]
    width = x.shape[3]
    out_channels = weight.shape[0]
    BLOCK_SIZE = 128  # Tunable parameter for block size

    # Determine the number of blocks needed
    grid = lambda meta: ((n_batch * out_channels * height * width + meta["BLOCK_SIZE"] - 1) // meta["BLOCK_SIZE"],)

    # Launch the Triton kernel
    pointwise_conv2d_kernel[grid](x, weight, out, n_batch, n_channels, height, width, out_channels, BLOCK_SIZE=BLOCK_SIZE)
    return out


class ModelNew(nn.Module):
    def __init__(self, in_channels: int, out_channels: int, kernel_size: int, stride: int = 1, padding: int = 0, dilation: int = 1, bias: bool = False):
        super(ModelNew, self).__init__()
        self.depthwise_weight = nn.Parameter(torch.randn(in_channels, kernel_size, kernel_size, device='cuda'))
        self.pointwise_weight = nn.Parameter(torch.randn(out_channels, in_channels, device='cuda'))
        
    def forward(self, x: torch.Tensor) -> torch.Tensor:
        """
        Performs the depthwise-separable 2D convolution.

        Args:
            x (torch.Tensor): Input tensor of shape (batch_size, in_channels, height, width).

        Returns:
            torch.Tensor: Output tensor of shape (batch_size, out_channels, height_out, width_out).
        """
        x = triton_depthwise_conv2d(x, self.depthwise_weight)
        x = triton_pointwise_conv2d(x, self.pointwise_weight)
        return x
