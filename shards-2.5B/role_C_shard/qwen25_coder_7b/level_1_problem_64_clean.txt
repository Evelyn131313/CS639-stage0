import torch
import torch.nn as nn
import triton
import triton.language as tl


@triton.jit
def conv1d_transpose_kernel(
    x_ptr,  # Pointer to input tensor
    weight_ptr,  # Pointer to weight tensor
    out_ptr,  # Pointer to output tensor
    n_batch,  # Number of batches
    n_channels,  # Number of input channels
    n_length,  # Length of the input signal
    kernel_size,  # Size of the convolution kernel
    stride,  # Stride of the convolution
    padding,  # Padding applied to the input
    BLOCK_SIZE_X: tl.constexpr,  # Block size in the X dimension
    BLOCK_SIZE_Y: tl.constexpr,  # Block size in the Y dimension,
):
    # Calculate the global indices
    x_idx = tl.program_id(0)
    w_idx = tl.program_id(1)
    
    # Calculate the local indices
    x_local_x = tl.arange(0, BLOCK_SIZE_X)
    x_local_y = tl.arange(0, BLOCK_SIZE_Y)
    
    # Create masks to ensure we don't go out of bounds
    x_mask = x_idx < n_batch
    w_mask = w_idx < n_channels
    
    # Load input and weight values
    x_values = tl.load(x_ptr + (x_idx * n_channels * n_length + w_idx * n_length + x_local_x), mask=x_mask, other=0.0)
    w_values = tl.load(weight_ptr + (w_idx * kernel_size + x_local_y), mask=w_mask, other=0.0)
    
    # Initialize output value
    out_value = 0.0
    
    # Perform the convolution
    for k in range(kernel_size):
        out_value += x_values * w_values[k]
    
    # Store the result
    tl.store(out_ptr + (x_idx * n_channels * n_length + w_idx * n_length + x_local_x), out_value, mask=x_mask)


def triton_conv1d_transpose(x: torch.Tensor, weight: torch.Tensor):
    """
    This function wraps the Triton kernel call. It:
      1. Ensures the inputs are contiguous on GPU.
      2. Calculates the grid (blocks) needed.
      3. Launches the Triton kernel.
    """
    assert x.is_cuda and weight.is_cuda, "Tensors must be on CUDA."
    x = x.contiguous()
    weight = weight.contiguous()

    # Prepare output tensor
    out = torch.empty_like(x)

    # Number of elements in the tensor
    n_batch = x.shape[0]
    n_channels = x.shape[1]
    n_length = x.shape[2]
    kernel_size = weight.shape[2]
    stride = 1  # Assuming stride is always 1 for simplicity
    padding = 0  # Assuming padding is always 0 for simplicity
    
    BLOCK_SIZE_X = 32  # Tunable parameter for block size in the X dimension
    BLOCK_SIZE_Y = 32  # Tunable parameter for block size in the Y dimension

    # Determine the number of blocks needed
    grid = lambda meta: ((n_batch + meta["BLOCK_SIZE_X"] - 1) // meta["BLOCK_SIZE_X"], (n_channels + meta["BLOCK_SIZE_Y"] - 1) // meta["BLOCK_SIZE_Y"],)

    # Launch the Triton kernel
    conv1d_transpose_kernel[grid](x, weight, out, n_batch, n_channels, n_length, kernel_size, stride, padding, BLOCK_SIZE_X=BLOCK_SIZE_X, BLOCK_SIZE_Y=BLOCK_SIZE_Y)
    return out


class ModelNew(nn.Module):
    def __init__(self, in_channels: int, out_channels: int, kernel_size: int, stride: int = 1, padding: int = 0, output_padding: int = 0, groups: int = 1, bias: bool = False):
        super(ModelNew, self).__init__()
        self.weight = nn.Parameter(torch.randn(out_channels, in_channels, kernel_size).cuda())
        
    def forward(self, x: torch.Tensor) -> torch.Tensor:
        """
        Performs the transposed 1D convolution.

        Args:
            x (torch.Tensor): Input tensor of shape (batch_size, in_channels, length).

        Returns:
            torch.Tensor: Output tensor of shape (batch_size, out_channels, length_out).
        """
        return triton_conv1d_transpose(x, self.weight)
