import torch
import torch.nn as nn
import triton
import triton.language as tl


@triton.jit
def conv3d_kernel(
    x_ptr,  # Pointer to input tensor
    w_ptr,  # Pointer to weight tensor
    y_ptr,  # Pointer to output tensor
    x_shape,  # Shape of the input tensor (batch_size, in_channels, depth, width, height)
    w_shape,  # Shape of the weight tensor (out_channels, in_channels, kernel_size, kernel_size, kernel_size)
    y_shape,  # Shape of the output tensor (batch_size, out_channels, depth_out, width_out, height_out)
    stride,  # Stride of the convolution
    padding,  # Padding applied to the input
    kernel_size,  # Size of the square convolution kernel
    BLOCK_SIZE_X: tl.constexpr,
    BLOCK_SIZE_Y: tl.constexpr,
    BLOCK_SIZE_Z: tl.constexpr,
):
    # Each program handles a contiguous block of data of size BLOCK_SIZE
    x_idx = tl.program_id(0)
    w_idx = tl.program_id(1)
    y_idx = tl.program_id(2)

    x_depth_out = y_shape[2]
    x_width_out = y_shape[3]
    x_height_out = y_shape[4]

    # Calculate the output indices
    x_d = x_idx // (x_width_out * x_height_out)
    x_w = (x_idx // x_height_out) % x_width_out
    x_h = x_idx % x_height_out

    # Calculate the input indices
    x_d_start = x_d * stride - padding
    x_w_start = x_w * stride - padding
    x_h_start = x_h * stride - padding

    # Calculate the weight indices
    w_d_start = 0
    w_w_start = 0
    w_h_start = 0

    # Initialize the output value
    out = 0.0

    # Iterate over the kernel
    for d in range(kernel_size):
        for w in range(kernel_size):
            for h in range(kernel_size):
                # Calculate the input indices for the current kernel element
                x_d_in = x_d_start + d
                x_w_in = x_w_start + w
                x_h_in = x_h_start + h

                # Ensure the input indices are within bounds
                mask = (x_d_in >= 0) & (x_d_in < x_shape[2]) & (x_w_in >= 0) & (x_w_in < x_shape[3]) & (x_h_in >= 0) & (x_h_in < x_shape[4])
                x_val = tl.load(x_ptr + ((x_d_in * x_shape[3] + x_w_in) * x_shape[4] + x_h_in) * x_shape[1] + w_idx, mask=mask, other=0.0)

                # Load the weight value for the current kernel element
                w_val = tl.load(w_ptr + ((d * kernel_size + w) * kernel_size + h) * w_shape[1] + x_idx, mask=mask, other=0.0)

                # Accumulate the result
                out += x_val * w_val

    # Store the result
    tl.store(y_ptr + x_idx, out)


def triton_conv3d(x: torch.Tensor, w: torch.Tensor, stride: int, padding: int, kernel_size: int):
    """
    This function wraps the Triton kernel call. It:
      1. Ensures the inputs are contiguous on GPU.
      2. Calculates the grid (blocks) needed.
      3. Launches the Triton kernel.
    """
    assert x.is_cuda and w.is_cuda, "Tensors must be on CUDA."
    x = x.contiguous()
    w = w.contiguous()

    # Prepare output tensor
    batch_size = x.shape[0]
    in_channels = x.shape[1]
    out_channels = w.shape[0]
    depth = x.shape[2]
    width = x.shape[3]
    height = x.shape[4]
    depth_out = (depth + 2 * padding - kernel_size) // stride + 1
    width_out = (width + 2 * padding - kernel_size) // stride + 1
    height_out = (height + 2 * padding - kernel_size) // stride + 1

    out = torch.empty((batch_size, out_channels, depth_out, width_out, height_out), device=x.device, dtype=x.dtype)

    # Number of elements in the tensor
    n_elements = batch_size * out_channels * depth_out * width_out * height_out
    BLOCK_SIZE_X = 32
    BLOCK_SIZE_Y = 32
    BLOCK_SIZE_Z = 1

    # Determine the number of blocks needed
    grid = lambda meta: (
        (depth_out + meta["BLOCK_SIZE_Z"] - 1) // meta["BLOCK_SIZE_Z"],
        (width_out + meta["BLOCK_SIZE_Y"] - 1) // meta["BLOCK_SIZE_Y"],
        (height_out + meta["BLOCK_SIZE_X"] - 1) // meta["BLOCK_SIZE_X"],
    )

    # Launch the Triton kernel
    conv3d_kernel[grid](x, w, out, x.shape, w.shape, out.shape, stride, padding, kernel_size, BLOCK_SIZE_X=BLOCK_SIZE_X, BLOCK_SIZE_Y=BLOCK_SIZE_Y, BLOCK_SIZE_Z=BLOCK_SIZE_Z)
    return out


class ModelNew(nn.Module):
    def __init__(self, in_channels: int, out_channels: int, kernel_size: int, stride: int = 1, padding: int = 0, dilation: int = 1, groups: int = 1, bias: bool = False):
        super(ModelNew, self).__init__()
        self.conv3d_weight = nn.Parameter(torch.randn(out_channels, in_channels, kernel_size, kernel_size, kernel_size, device="cuda"))
        
    def forward(self, x: torch.Tensor) -> torch.Tensor:
        """
        Performs the 3D convolution using Triton kernel.

        Args:
            x (torch.Tensor): Input tensor of shape (batch_size, in_channels, depth, width, height).

        Returns:
            torch.Tensor: Output tensor of shape (batch_size, out_channels, depth_out, width_out, height_out).
        """
        return triton_conv3d(x, self.conv3d_weight, stride, padding, kernel_size)
