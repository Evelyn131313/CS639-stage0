import torch
import torch.nn as nn
import triton
import triton.language as tl

@triton.jit
def matmul_kernel(
    A_ptr, B_ptr, C_ptr,
    M, N, K,
    BLOCK_SIZE: tl.constexpr,
):
    # Each thread block computes a block of the output matrix
    i_block = tl.program_id(0)
    j_block = tl.program_id(1)
    i = i_block * BLOCK_SIZE + tl.program_id(0)
    j = j_block * BLOCK_SIZE + tl.program_id(1)

    # Shared memory for A and B tiles
    A_shared = tl.shared(float32[BLOCK_SIZE, K])
    B_shared = tl.shared(float32[BLOCK_SIZE, N])

    # Load the tile of A into shared memory
    for k in range(K):
        A_shared[tl.program_id(0), k] = tl.load(A_ptr + i * K + k)

    # Load the tile of B into shared memory
    for k in range(K):
        B_shared[k, tl.program_id(1)] = tl.load(B_ptr + k * N + j)

    # Compute the dot product
    acc = 0.0
    for k in range(K):
        acc += A_shared[tl.program_id(0), k] * B_shared[k, tl.program_id(1)]

    # Store the result
    tl.store(C_ptr + i * N + j, acc)

def triton_matmul(A: torch.Tensor, B: torch.Tensor):
    assert A.is_cuda and B.is_cuda, "Tensors must be on CUDA."
    A = A.contiguous()
    B = B.contiguous()
    C = torch.empty(A.shape[0], B.shape[1], device=A.device, dtype=A.dtype)

    M = A.shape[0]
    N = B.shape[1]
    K = A.shape[1]

    # Determine the block size
    BLOCK_SIZE = 128

    # Compute the grid size
    grid_x = (M + BLOCK_SIZE - 1) // BLOCK_SIZE
    grid_y = (N + BLOCK_SIZE - 1) // BLOCK_SIZE

    # Launch the kernel
    matmul_kernel[grid_x, grid_y](A, B, C, M, N, K, BLOCK_SIZE=BLOCK_SIZE)
    return C

class ModelNew(nn.Module):
    def __init__(self):
        super(ModelNew, self).__init__()

    def forward(self, A: torch.Tensor, B: torch.Tensor) -> torch.Tensor:
        return triton_matmul(A, B)
